018 package builders.dsl.spreadsheet.query.poi;
019 
020 import builders.dsl.spreadsheet.api.*;
021 import org.apache.poi.xssf.usermodel.XSSFCellStyle;
022 import org.apache.poi.xssf.usermodel.extensions.XSSFCellBorder;
023 
024 class PoiCellStyle implements CellStyle {
025     PoiCellStyle(XSSFCellStyle style) {
026         this.style = style;
027     }
028 
029     @Override
030     public Color getForeground() {
031         if (style.getFillForegroundColorColor() == null) {
032             return null;
033         }
034         if (style.getFillForegroundColorColor().getRGB() == null) {
035             return null;
036         }
037         return new Color(style.getFillForegroundColorColor().getRGB());
038     }
039 
040     @Override
041     public Color getBackground() {
042         if (style.getFillBackgroundColorColor() == null) {
043             return null;
044         }
045         if (style.getFillBackgroundColorColor().getRGB() == null) {
046             return null;
047         }
048         return new Color(style.getFillBackgroundColorColor().getRGB());
049     }
050 
051     @Override
052     public ForegroundFill getFill() {
053         switch (style.getFillPattern()) {
054             case NO_FILL:
055                 return ForegroundFill.NO_FILL;
056             case SOLID_FOREGROUND:
057                 return ForegroundFill.SOLID_FOREGROUND;
058             case FINE_DOTS:
059                 return ForegroundFill.FINE_DOTS;
060             case ALT_BARS:
061                 return ForegroundFill.ALT_BARS;
062             case SPARSE_DOTS:
063                 return ForegroundFill.SPARSE_DOTS;
064             case THICK_HORZ_BANDS:
065                 return ForegroundFill.THICK_HORZ_BANDS;
066             case THICK_VERT_BANDS:
067                 return ForegroundFill.THICK_VERT_BANDS;
068             case THICK_BACKWARD_DIAG:
069                 return ForegroundFill.THICK_BACKWARD_DIAG;
070             case THICK_FORWARD_DIAG:
071                 return ForegroundFill.THICK_FORWARD_DIAG;
072             case BIG_SPOTS:
073                 return ForegroundFill.BIG_SPOTS;
074             case BRICKS:
075                 return ForegroundFill.BRICKS;
076             case THIN_HORZ_BANDS:
077                 return ForegroundFill.THIN_HORZ_BANDS;
078             case THIN_VERT_BANDS:
079                 return ForegroundFill.THIN_VERT_BANDS;
080             case THIN_BACKWARD_DIAG:
081                 return ForegroundFill.THIN_BACKWARD_DIAG;
082             case THIN_FORWARD_DIAG:
083                 return ForegroundFill.THIN_FORWARD_DIAG;
084             case SQUARES:
085                 return ForegroundFill.SQUARES;
086             case DIAMONDS:
087                 return ForegroundFill.DIAMONDS;
088         }
089         return null;
090     }
091 
092     @Override
093     public int getIndent() {
094         return style.getIndention();
095     }
096 
097     @Override
098     public int getRotation() {
099         return style.getRotation();
100     }
101 
102     @Override
103     public String getFormat() {
104         return style.getDataFormatString();
105     }
106 
107     @Override
108     public Font getFont() {
109         return style.getFont() != null ? new PoiFont(style.getFont()) : null;
110     }
111 
112     @Override
113     public Border getBorder(Keywords.BorderSide borderSide) {
114         if (Keywords.BorderSide.TOP.equals(borderSide)) {
115             return new PoiBorder(style.getBorderColor(XSSFCellBorder.BorderSide.TOP), style.getBorderTopEnum());
116         } else if (Keywords.BorderSide.BOTTOM.equals(borderSide)) {
117             return new PoiBorder(style.getBorderColor(XSSFCellBorder.BorderSide.BOTTOM), style.getBorderBottomEnum());
118         } else if (Keywords.BorderSide.LEFT.equals(borderSide)) {
119             return new PoiBorder(style.getBorderColor(XSSFCellBorder.BorderSide.LEFT), style.getBorderLeftEnum());
120         } else if (Keywords.BorderSide.RIGHT.equals(borderSide)) {
121             return new PoiBorder(style.getBorderColor(XSSFCellBorder.BorderSide.RIGHT), style.getBorderRightEnum());
122         }
123         return new PoiBorder(null, null);
124     }
125 
126     private final XSSFCellStyle style;
127 }
