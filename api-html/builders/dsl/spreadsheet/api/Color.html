168     private final String hex;
169 
170     public Color(String hex) {
171         if (!hex.matches("#[\\dA-Fa-f]{6}")) {
172             throw new IllegalArgumentException("Wrong format for color: " + hex);
173         }
174         this.hex = hex.toUpperCase();
175     }
176 
177     public Color(byte[] rgb) {
178         if (rgb.length != 3) {
179             throw new IllegalArgumentException("Wrong number of parts in: " + Arrays.toString(rgb));
180         }
181         this.hex = String.format("#%02X%02X%02X", rgb[0], rgb[1], rgb[2]);
182     }
183 
184     @Override
185     public boolean equals(Object o) {
186         if (this == o) {
187             return true;
188         }
189         if (o == null || getClass() != o.getClass()) {
190             return false;
191         }
192 
193         Color color = (Color) o;
194 
195         return !(hex != null ? !hex.equals(color.hex) : color.hex != null);
196 
197     }
198 
199     @Override
200     public int hashCode() {
201         return hex != null ? hex.hashCode() : 0;
202     }
203 
204     public String getHex() {
205         return hex;
206     }
207 
208     @Override
209     public String toString() {
210         return "Color[" + getHex() + "]";
211     }
